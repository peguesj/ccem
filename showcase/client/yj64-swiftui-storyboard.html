RomBrowserView — Annotated SwiftUI Pseudo-code
var body: some View {
  VStack(spacing: 0) {
    // ── Toolbar ──────────────────────────────────────────────────────
    HStack {
      Picker("View", selection: $browserMode) { ... }
        .disabled(scanner.isScanning)           // State A: picker locked

      if scanner.isScanning {                   // State A branch
        ProgressView()
      }

      Text("\(filteredRoms.count) ROMs")

      TextField("Search", text: $searchText)

      Button("Reload", action: reload)
        .disabled(scanner.isScanning)
    }

    // ── Content Area ─────────────────────────────────────────────────
    ZStack {
      // Layer 1: last-known content (always rendered)
      if !scanner.isScanning && filteredRoms.isEmpty && searchText.isEmpty {
        EmptyRomsView()                              // State B
      } else if !scanner.isScanning && filteredRoms.isEmpty {
        EmptySearchView(query: searchText)          // State C
      } else if browserMode == .grid {
        RomGridView(roms: filteredRoms)             // State D
      } else {
        RomListView(roms: filteredRoms)             // State E
      }

      // Layer 2: scanning overlay on top
      if scanner.isScanning {
        ScanningOverlayView()                        // State A overlay
          .background(.ultraThinMaterial)
      }
    }
  }
}
